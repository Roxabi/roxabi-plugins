  import { readFileSync, writeFileSync, mkdirSync, existsSync } from 'node:fs'
  export interface WorkspaceProject { repo: string; projectId: string; label: string }
  export interface Workspace { projects: WorkspaceProject[] }

  export function getWorkspacePath(): string {
    const vault = `${process.env.HOME}/.roxabi-vault`
    if (existsSync(vault)) return `${vault}/workspace.json`
    return `${process.env.HOME}/.config/roxabi/workspace.json`
  }

  export function readWorkspace(): Workspace {
    const p = getWorkspacePath()
    if (!existsSync(p)) return { projects: [] }
    return JSON.parse(readFileSync(p, 'utf8')) as Workspace
  }

  export function writeWorkspace(ws: Workspace): void {
    const p = getWorkspacePath()
    const dir = p.substring(0, p.lastIndexOf('/'))
    mkdirSync(dir, { recursive: true, mode: 0o700 })
    writeFileSync(p, JSON.stringify(ws, null, 2) + '\n')
  }

  /** Build a batched multi-project query with N aliased node() lookups.
   *  Response shape: data.project0, data.project1, ...
   */
  export function buildBatchedQuery(projectIds: string[]): string {
    const fragments = projectIds.map((_, i) => `
      project${i}: node(id: $project${i}Id) {
        ... on ProjectV2 {
          items(first: 100) {
            pageInfo { hasNextPage endCursor }
            nodes { /* same shape as ISSUES_QUERY node fragment */ }
          }
        }
      }`)
    const params = projectIds.map((_, i) => `$project${i}Id: ID!`).join(', ')
    return `query(${params}) { ${fragments.join('')} }`
  }
  ```

  import type { WorkspaceProject } from '../../../cli/lib/workspace'  // or inline type
  /** Fetch items for all workspace projects in a single batched GraphQL request. */
  export async function fetchAllProjects(
    projects: WorkspaceProject[]
  ): Promise<Map<string, RawItem[]>> {
    if (projects.length === 0) return new Map()
    const query = buildBatchedQuery(projects.map(p => p.projectId))
    const variables = Object.fromEntries(projects.map((p, i) => [`project${i}Id`, p.projectId]))
    const data = await ghGraphQL(query, variables)
    const result = new Map<string, RawItem[]>()
    for (let i = 0; i < projects.length; i++) {
      const node = (data as any).data[`project${i}`]
      result.set(projects[i].label, node?.items?.nodes ?? [])
    }
    return result
  }
  // Keep fetchAllItems() for backward compat (single-project mode)

  import { readWorkspace } from '../../../cli/lib/workspace'
  import { detectGitHubRepo } from '../../shared/config'

  /** Resolve project config: workspace.json entry for current repo → .env fallback. */
  export function resolveConfig(): { projectId: string; source: 'workspace' | 'env' } {
    const ws = readWorkspace()
    const currentRepo = detectGitHubRepo()
    const entry = ws.projects.find(p => p.repo === currentRepo)
    if (entry) return { projectId: entry.projectId, source: 'workspace' }
    const envId = process.env.PROJECT_ID ?? ''
    if (envId) return { projectId: envId, source: 'env' }
    return { projectId: '', source: 'env' }
  }
  ```
