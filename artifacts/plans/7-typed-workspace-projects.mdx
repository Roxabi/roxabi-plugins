  ```typescript
  import type { WorkspaceProject, ProjectFieldIds } from './workspace'

  export function resolveFieldIds(project: WorkspaceProject): ProjectFieldIds {
    if (project.fieldIds) {
      if (!project.fieldIds.status) {
        throw new Error(`[project ${project.label}] fieldIds.status is required`)
      }
      return project.fieldIds  // complete source — no env mixing
    }
    // fallback: build from .env (existing behavior)
    return {
      status: STATUS_FIELD_ID,
      col2: SIZE_FIELD_ID,
      col3: PRIORITY_FIELD_ID,
      statusOptions: STATUS_OPTIONS,
      col2Options: SIZE_OPTIONS,
      col3Options: PRIORITY_OPTIONS,
    }
  }

  export function fieldIdForSlot(
    project: WorkspaceProject,
    slot: 'col2' | 'col3'
  ): string | undefined {
    return resolveFieldIds(project)[slot]
  }
  ```

  ```typescript
  async function resolveProjectFieldIds(projectId: string, type: ProjectType): Promise<ProjectFieldIds> {
    const fields = await queryGitHubProjectFields(projectId)
    const SLOT_NAMES = type === 'company'
      ? { col2: ['Quarter'], col3: ['Pillar'] }
      : { col2: ['Size'], col3: ['Priority'] }
    const col2Field = fields.find(f => SLOT_NAMES.col2.includes(f.name))
    const col3Field = fields.find(f => SLOT_NAMES.col3.includes(f.name))
    const statusField = fields.find(f => f.name === 'Status')
    if (!statusField || !col2Field || !col3Field) {
      console.warn(`[init] some fields not found for project ${projectId}; writing fieldIds: {}`)
      return {}  // partial or empty — user fills manually
    }
    return {
      status: statusField.id, col2: col2Field.id, col3: col3Field.id,
      statusOptions: toOptionMap(statusField), col2Options: toOptionMap(col2Field),
      col3Options: toOptionMap(col3Field),
    }
  }
  ```
